cmake_minimum_required(VERSION 4.0.0)

project(CinderWolfeEngine CXX)

set(CMAKE_CXX_SCAN_FOR_MODULES 1)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include(FetchContent)

# GLFW
FetchContent_Declare(
    glfw
    GIT_REPOSITORY https://github.com/glfw/glfw.git
    GIT_TAG 3.4
)
FetchContent_MakeAvailable(glfw)

add_executable(${PROJECT_NAME})

# Source Files
target_sources(${PROJECT_NAME}
    PRIVATE 
    "${CMAKE_CURRENT_SOURCE_DIR}/core/main.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/graphics/opengl/glad.c"
)

# Sources for c++ Modules
target_sources(${PROJECT_NAME} 
 PRIVATE
 FILE_SET cxx_modules 
 TYPE CXX_MODULES 
 FILES
 "${CMAKE_CURRENT_SOURCE_DIR}/math/CinderWolfeEngine.Math.cppm"
 "${CMAKE_CURRENT_SOURCE_DIR}/core/CinderWolfeEngine.Engine.cppm"
 "${CMAKE_CURRENT_SOURCE_DIR}/graphics/CinderWolfeEngine.Window.cppm")

 target_include_directories(${PROJECT_NAME}
    PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/graphics/opengl
)

# -----------------------------
# Libraries
# -----------------------------
target_link_libraries(${PROJECT_NAME}
    PRIVATE
        glfw
)

 # Find open gl package on all platforms
 find_package(OpenGL REQUIRED)
 target_link_libraries(${PROJECT_NAME} PRIVATE OpenGL::GL)
